\newcommand{\zaklinadlo}[5][]{{%
    \setlength{\tabcolsep}{\zaklinadlotabcolsep}%
    % \radekzaklinadla{symbol}{text}
    \newcommand{\radekzaklinadla}[2]{%
        \nepoprve{\par}%
        % strut (vertikální roztažení řádku)
        %%\rule{0pt}{1.5ex}%
        % symbol řádku (vykreslí se před sloupec)
        \makebox[0pt][r]{{##1}~}%
        % text řádku (může dojít k zalomení)
        {##2}%
        }%
    \noindent\begin{tabular}{%
        % číslo zaklínadla:
        >{\raggedright\fontsize{9pt}{0pt}\selectfont}p{\sirkacislazaklinadla}@{}%
        % ikona zaklínadla:
        >{\centering}p{\sirkaikony}@{\hspace{2pt}}%
        % titulek zaklínadla + mezera před řádky zaklínadla:
        >{\raggedright\fontsize{10pt}{10pt}\selectfont}p{\sirkatitulkuzaklinadla}%
        @{\hspace{10pt}}%
        % řádky zaklínadla:
        |>{\raggedright\zaklinadlofamily\zaklinadlosize}p{\sirkaradkuzaklinadla}%
    }%
    % číslo zaklínadla:
    \##2%
    % ikona zaklínadla:
    &{\lmmathfamily{}♣}%
    % titulek zaklínadla:
    &{#3}%
    % řádky zaklínadla:
    &{\raggedright\budepoprve{}#5}%
    \tabularnewline%
    \end{tabular}#4\par\vspace{4pt plus5pt minus5pt}%
    }}
